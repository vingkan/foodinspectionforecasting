def group_crits_month(df_data):
    # Calculate proportion of inspections with critical found / all inspections per month
    n_insps_month = df_data.groupby(by="month")["month"].count().values
    n_found_month = df_data.groupby(by="month")[TARGET].sum().values
    p_found_month = n_found_month / n_insps_month
    # Calculate proportion of each violation / all instances of critical violations per month
    n_viols_month = df_data.groupby(by="month")[VIOLATION_LABELS].sum().values
    n_crits_month = n_viols_month.sum(axis=1)
    n_totals_month = np.vstack([n_crits_month for v in VIOLATION_LABELS]).T
    p_viols_month = n_viols_month / n_totals_month
    res_found = (n_insps_month, n_found_month, p_found_month)
    res_viols = (n_viols_month, n_totals_month, p_viols_month)
    return res_found, res_viols

def date_df_by_score(df_data, crit_mat, scores):
    SECONDS = 60 * 60 * 24
    idf = pd.DataFrame({
        "Inspection_ID": df_data[INSPECTION_ID],
        "new_inspection_date": get_days_by_score(df_data[DATE], scores),
        "score": scores,
        "criticalFound": df_data[TARGET].astype(int)
    })
    calendar_date = idf["new_inspection_date"].apply(lambda d: datetime.fromtimestamp((d + 1) * SECONDS))
    idf["year"] = calendar_date.apply(lambda d: d.strftime("%Y"))
    idf["month"] = calendar_date.apply(lambda d: d.strftime("%m"))
    idf = idf.set_index(INSPECTION_ID).join(crit_mat.set_index(INSPECTION_ID))
    idf = idf.reset_index()
    return idf
